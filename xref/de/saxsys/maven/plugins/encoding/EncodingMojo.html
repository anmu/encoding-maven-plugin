79  	 * {@inheritDoc}
80  	 * @see org.apache.maven.plugin.AbstractMojo#execute()
81  	 */
82  	public void execute() throws MojoExecutionException, MojoFailureException {
83  
84  		if (!encodingIsValid(encoding)) {
85  			throw new MojoExecutionException(bundle.getString(Constants.RB_ENCODING_ERROR_NO_SUPPORT));
86  		}
87  		getLog().info(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_USAGE_ENCODING), encoding));
88  
89  		if (!fileSetEncodingIsValid(fileSet)) {
90  			getLog().error(bundle.getString(Constants.RB_ENCODING_ERROR_NO_SUPPORT_OF_FILESET));
91  			getLog().error("");
92  
93  			throw new MojoExecutionException(bundle.getString(Constants.RB_ENCODING_ERROR_NO_SUPPORT));
94  		}
95  		getLog().info("");
96  
97  		Worker2Configuration workerConfiguration = new Worker2Configuration(encoding, fileSet);
98  		Worker2 worker = new Worker2(workerConfiguration);
99  
100 		// Source-Directory
101 		String srcFolder = project.getBuild().getSourceDirectory();
102 		if (!StringUtils.isEmpty(srcFolder)) {
103 			FolderSet srcFolderSet = worker.work("SourceFolder", srcFolder);
104 			getLog().info(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_CHECKING_DIRECTORY), srcFolderSet.getFolderCaption()));
105 			long correctFiles = 0;
106 			for (Entry entry : srcFolderSet.getEntries()) {
107 				displayLog(entry);
108 				if (entry.getState().equals(State.CORRECT_ENCODING)) {
109 					correctFiles++;
110 				}
111 			}
112 			displayFooterIfAllFilesCorrect(srcFolderSet, correctFiles);
113 			getLog().info("");
114 		}
115 
116 		// Test-Source-Directory
117 		String srcTestFolder = project.getBuild().getTestSourceDirectory();
118 		if (!StringUtils.isEmpty(srcTestFolder)) {
119 			FolderSet srcTestFolderSet = worker.work("TestSourceFolder", srcTestFolder);
120 			getLog().info(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_CHECKING_DIRECTORY), srcTestFolderSet.getFolderCaption()));
121 			long correctFiles = 0;
122 			for (Entry entry : srcTestFolderSet.getEntries()) {
123 				displayLog(entry);
124 				if (entry.getState().equals(State.CORRECT_ENCODING)) {
125 					correctFiles++;
126 				}
127 			}
128 			displayFooterIfAllFilesCorrect(srcTestFolderSet, correctFiles);
129 			getLog().info("");
130 		}
131 
132 		// Resource-Directory
133 		for (Resource resource : project.getBuild().getResources()) {
134 			String resourceFolder = resource.getDirectory();
135 			if (!StringUtils.isEmpty(resourceFolder)) {
136 				FolderSet resourceFolderSet = worker.work("ResourceFolder", resourceFolder);
137 				getLog().info(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_CHECKING_DIRECTORY), resourceFolderSet.getFolderCaption()));
138 				long correctFiles = 0;
139 				for (Entry entry : resourceFolderSet.getEntries()) {
140 					displayLog(entry);
141 					if (entry.getState().equals(State.CORRECT_ENCODING)) {
142 						correctFiles++;
143 					}
144 				}
145 				displayFooterIfAllFilesCorrect(resourceFolderSet, correctFiles);
146 				getLog().info("");
147 			}
148 		}
149 
150 		if (!allFilesCorrect && failsOnError) {
151 			throw new MojoExecutionException(bundle.getString(Constants.RB_ENCODING_MESSAGE_EXCEPTION));
152 		}
153 	}

162 	/**
163 	 * Checks the general encoding is correct.
164 	 * @param _encoding The encoding as {@link String}.
165 	 * @return Returns <code>true</code> if it is correct, otherwise <code>false</code>.
166 	 */
167 	private boolean encodingIsValid(final String _encoding) {
168 
169 		boolean foundCharset = false;
170 		String supportedCharsets = "";
171 
172 		StringBuffer buf = new StringBuffer();
173 		for (String charset : CharsetDetector.getAllDetectableCharsets()) {
174 			if (charset.equalsIgnoreCase(_encoding)) {
175 				foundCharset = true;
176 			}
177 			buf.append(", " + charset);
178 //			supportedCharsets = supportedCharsets + ", " + charset;
179 		}
180 		supportedCharsets = buf.toString().substring(2);
181 
182 		if (!foundCharset) {
183 			getLog().error(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_ICU_NO_SUPPORT), _encoding));
184 			getLog().error(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_ICU_SUPPORTED_ENCODINGS), supportedCharsets));
185 			getLog().error("");
186 		}
187 
188 		return foundCharset;
189 	}

191 	private boolean fileSetEncodingIsValid(Map<String, String> _fileSet) {
192 
193 		boolean encodingIsSupported = true;
194 
195 		for (String fileExtension : fileSet.keySet()) {
196 
197 			String fileExtensionEncoding = fileSet.get(fileExtension);
198 
199 			if (StringUtils.isEmpty(fileExtensionEncoding)) {
200 				fileExtensionEncoding = Constants.DEFAULT_ENCODING;
201 			}
202 
203 			if (!encodingIsValid(fileExtensionEncoding)) {
204 				encodingIsSupported = false;
205 				return encodingIsSupported;
206 			}
207 
208 			getLog().info(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_USAGE_ENCODING_FILETYPE), fileExtensionEncoding, fileExtension));
209 		}
210 
211 		return encodingIsSupported;
212 	}

215 
216 		if (_entry.getState() == State.CORRECT_ENCODING) {
217 			getLog().debug(
218 					"  " + _entry.getReadablePath() + " [" + _entry.getEncoding()
219 							+ "] -> OK");
220 		} else if (_entry.getState() == State.NOTHING_DETECTED) {
221 			getLog().debug(
222 					"  " + _entry.getReadablePath() + " [" + _entry.getEncoding()
223 							+ "] -> OK");
224 		} else {
225 			getLog().error(
226 					"  " + _entry.getReadablePath() + " [" + _entry.getEncoding()
227 							+ "] -> ERROR");
228 			this.allFilesCorrect = false;
229 		}
