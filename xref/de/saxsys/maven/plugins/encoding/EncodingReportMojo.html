284 	 * Calculates a floating value two of decimal places.
285 	 * 
286 	 * @param sum Amount of files.
287 	 * @param value Amount of file from a given state.
288 	 * 
289 	 * @return Returns a floating value of two decimal places.
290 	 */
291 	private Float percentage(Long sum, Long value) {
292 		Float percentage = (float) (100 * (float)value / (float)sum);
293 		percentage = percentage * 100;
294 		percentage = (float) Math.round(percentage);
295 		percentage = percentage / 100;
296 		return percentage;
297 	}

300 	 * 
301 	 * 
302 	 * @param state
303 	 * @return The amount of files.
304 	 */
305 	private Long count(State state) {
306 		
307 		Long value = Long.valueOf(0);
308 		
309 		for (Entry entry : getSourceFolderSet().getEntries()) {
310 			if (entry.getState().equals(state)) {
311 				value++;
312 			}
313 		}
314 		
315 		for (Entry entry : getTestSourceFolderSet().getEntries()) {
316 			if (entry.getState().equals(state)) {
317 				value++;
318 			}
319 		}
320 		
321 		for (FolderSet folderSet : getResourceFolderSet()) {
322 			for (Entry entry : folderSet.getEntries()) {
323 				if (entry.getState().equals(state)) {
324 					value++;
325 				}
326 			}
327 		}
328 
329 		return value;
330 	}

495 	/**
496 	 * Checks the general encoding is correct.
497 	 * 
498 	 * @param _encoding The encoding as {@link String}.
499 	 * 
500 	 * @return Returns <code>true</code> if it is correct, otherwise <code>false</code>.
501 	 */
502 	private boolean encodingIsValid(String _encoding) {
503 		
504 		boolean foundCharset = false;
505 		String supportedCharsets = "";
506 		
507 		for (String charset : CharsetDetector.getAllDetectableCharsets()) {
508 			if (charset.equalsIgnoreCase(_encoding)) {
509 				foundCharset = true;
510 			}
511 			supportedCharsets = supportedCharsets + ", " + charset;
512 		}
513 		supportedCharsets = supportedCharsets.substring(2);
514 
515 		if (!foundCharset) {
516 			getLog().error(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_ICU_NO_SUPPORT), _encoding));
517 			getLog().error(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_ICU_SUPPORTED_ENCODINGS), supportedCharsets));
518 			getLog().error("");
519 		}
520 
521 		return foundCharset;
522 	}

524 	private boolean fileSetEncodingIsValid(Map<String, String> _fileSet) {
525 
526 		boolean encodingIsSupported = true;
527 		
528 		for (String fileExtension : fileSet.keySet()) {
529 			
530 			String fileExtensionEncoding = fileSet.get(fileExtension);
531 			
532 			if (StringUtils.isEmpty(fileExtensionEncoding)) {
533 				fileExtensionEncoding = Constants.DEFAULT_ENCODING;
534 			}
535 			
536 			if (!encodingIsValid(fileExtensionEncoding)) {
537 				encodingIsSupported = false;
538 			}
539 
540 			getLog().debug(MessageFormat.format(bundle.getString(Constants.RB_ENCODING_MESSAGE_USAGE_ENCODING_FILETYPE), fileExtensionEncoding, fileExtension));
541 		}
542 
543 		return encodingIsSupported;
544 	}
